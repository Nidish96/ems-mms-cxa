In this file we derive the Fourier Coefficients (in rectangular form) of the reaction forces of a
Jenkins element under harmonic displacement. Expressions of higher harmonics are also
developed.
--> kill(all)$
τDst:acos(1−2·fs/(kt·a));
gradef(charfun(τ),0)$
assume(kt>0)$
assume(a>0)$
assume(fs>0)$
assume(τD>0)$

\[\]\[\tag{ensuremath{\tau}Ds} \pi \mathop{-}\operatorname{acos}{\left( \frac{2 \ensuremath{\mathrm{fs}}}{a\, \ensuremath{\mathrm{kt}}}\mathop{-}1\right) }\]

--> f1(τ):=kt·a·(cos(τ+β)−1)+fs$
f2(τ):=−fs$
f3(τ):=kt·a·(cos(τ+β)+1)−fs$
f4(τ):=fs$
--> Fint:factor(integrate(f1(τ)·exp(%i·τ), τ, −β, −β+τDst) +
integrate(f2(τ)·exp(%i·τ), τ, −β+τDst, −β+%pi) +
integrate(f3(τ)·exp(%i·τ), τ, −β+%pi, −β+%pi+τDst) +
integrate(f4(τ)·exp(%i·τ), τ, −β+%pi+τDst, −β+2·%pi))/%pi$
Fints:factor(exponentialize(trigreduce(subst([fs=kt·a/2·(1−cos(τst))], Fint))));

\[\]\[\tag{int} \frac{a\, \ensuremath{\mathrm{kt}} {{\% e}^{-\left( \% i \beta \right) }} \left( \% i {{\% e}^{2 \% i \ensuremath{\mathrm{\tau st}}}}\mathop{+}2 \ensuremath{\mathrm{\tau st}}\mathop{-}\% i\right) }{2 \pi }\]

--> Fc:factor(subst([cos(β)=Uc/a,sin(β)=−Us/a], realpart(Fints)));
Fs:factor(subst([cos(β)=Uc/a,sin(β)=−Us/a], imagpart(Fints)));

\[\]\[\tag{} \mathop{-}\left( \frac{\ensuremath{\mathrm{kt}}\, \left( \ensuremath{\mathrm{Uc}} \sin{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\mathop{+}\ensuremath{\mathrm{Us}} \cos{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\mathop{-}2 \ensuremath{\mathrm{Uc}}\, \ensuremath{\mathrm{\tau st}}\mathop{-}\ensuremath{\mathrm{Us}}\right) }{2 \pi }\right) \]

\[\]\[\tag{} \mathop{-}\left( \frac{\ensuremath{\mathrm{kt}}\, \left( \ensuremath{\mathrm{Us}} \sin{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\mathop{-}\ensuremath{\mathrm{Uc}} \cos{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\mathop{-}2 \ensuremath{\mathrm{Us}}\, \ensuremath{\mathrm{\tau st}}\mathop{+}\ensuremath{\mathrm{Uc}}\right) }{2 \pi }\right) \]

--> Fintn(n):=radcan(exponentialize(trigreduce(subst([fs=kt·a/2·(1−cos(τst))],
               integrate(f1(τ)·exp(%i·n·τ), τ, −β, −β+τDst) +
               integrate(f2(τ)·exp(%i·n·τ), τ, −β+τDst, −β+%pi) +
               integrate(f3(τ)·exp(%i·n·τ), τ, −β+%pi, −β+%pi+τDst) +
               integrate(f4(τ)·exp(%i·n·τ), τ, −β+%pi+τDst, −β+2·%pi))/%pi)))$
--> factor(Fintn(3));

\[\]\[\tag{%o16} \mathop{-}\left( \frac{\% i a\, \ensuremath{\mathrm{kt}} {{\% e}^{-\left( 3 \% i \beta \right) }} \left( {{\% e}^{\% i \ensuremath{\mathrm{\tau st}}}}\mathop{-}1\right) \, \left( {{\% e}^{\% i \ensuremath{\mathrm{\tau st}}}}\mathop{+}1\right) \, \left( 2 {{\% e}^{2 \% i \ensuremath{\mathrm{\tau st}}}}\mathop{+}1\right) }{12 \pi }\right) \]

--> factor(Fintn(5));

\[\]\[\tag{%o17} \mathop{-}\left( \frac{\% i a\, \ensuremath{\mathrm{kt}} {{\% e}^{-\left( 5 \% i \beta \right) }} \left( {{\% e}^{\% i \ensuremath{\mathrm{\tau st}}}}\mathop{-}1\right) \, \left( {{\% e}^{\% i \ensuremath{\mathrm{\tau st}}}}\mathop{+}1\right) \, \left( 3 {{\% e}^{4 \% i \ensuremath{\mathrm{\tau st}}}}\mathop{+}{{\% e}^{2 \% i \ensuremath{\mathrm{\tau st}}}}\mathop{+}1\right) }{60 \pi }\right) \]

--> jacobian([Fc,Fs],[Uc,Us]);

\[\]\[\tag{%o18} \begin{pmatrix}\mathop{-}\left( \frac{\ensuremath{\mathrm{kt}}\, \left( \sin{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\mathop{-}2 \ensuremath{\mathrm{\tau st}}\right) }{2 \pi }\right) & \mathop{-}\left( \frac{\ensuremath{\mathrm{kt}}\, \left( \cos{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\mathop{-}1\right) }{2 \pi }\right) \\ \mathop{-}\left( \frac{\ensuremath{\mathrm{kt}}\, \left( 1\mathop{-}\cos{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\right) }{2 \pi }\right) & \mathop{-}\left( \frac{\ensuremath{\mathrm{kt}}\, \left( \sin{\left( 2 \ensuremath{\mathrm{\tau st}}\right) }\mathop{-}2 \ensuremath{\mathrm{\tau st}}\right) }{2 \pi }\right) \end{pmatrix}\]

--> factor(Fintn(1));
